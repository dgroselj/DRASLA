189 void PS2DTurb::addForcing (fftw_complex *omegaSp0, double timeStep, bool randomPhases){
190  double noiseCorrect = 1.0/sqrt(timeStep);
191  double cosPhase, sinPhase, omegaAbs, randNum;
192  int arrayInd;
193 #pragma omp parallel for default(shared) private(cosPhase,sinPhase,omegaAbs,randNum,arrayInd)
194  for (int k = 0; k < forcedWaveNumbers.size(); k++){
195  arrayInd = forcedWaveNumbers[k];
196  omegaAbs = sqrt(pow(omegaSp0[arrayInd][0],2) + pow(omegaSp0[arrayInd][1],2));
197 #pragma omp critical
198  {
199  randNum = mtran.randExc();
200  }
201  if (randomPhases || omegaAbs/norm < numeric_limits<double>::epsilon()){
202  cosPhase = cos(2*PI*randNum);
203  sinPhase = sin(2*PI*randNum);
204  }
205  else {
206  cosPhase = ((randNum < 0.5) ? 1 : -1)*omegaSp0[arrayInd][1]/omegaAbs;
207  sinPhase = - ((randNum < 0.5) ? 1 : -1)*omegaSp0[arrayInd][0]/omegaAbs;
208  }
209  nonLinGridSp[arrayInd][0] += noiseCorrect*forceAmplitudes[k]*cosPhase;
210  nonLinGridSp[arrayInd][1] += noiseCorrect*forceAmplitudes[k]*sinPhase;
211  //Preserve Hermitian symmetry at Ky=0:
212  if (arrayInd % (nPoints/2+1) == 0){
213  nonLinGridSp[arraySizeSp - arrayInd][0] = nonLinGridSp[arrayInd][0];
214  nonLinGridSp[arraySizeSp - arrayInd][1] = - nonLinGridSp[arrayInd][1];
