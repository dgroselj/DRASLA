186 double DRASLA::interpolateCubic(double *grid1D, double s){
187  double d1, d2, d10, d20, val1, val2, delta, derivR;
188  val1 = grid1D[2];
189  val2 = grid1D[3];
190  d1 = (grid1D[0] - 8*grid1D[1] + 8*grid1D[3] - grid1D[4])/12;
191  d2 = (grid1D[1] - 8*grid1D[2] + 8*grid1D[4] - grid1D[5])/12;
192  d10 = (grid1D[3] - grid1D[1])/2;
193  d20 = (grid1D[4] - grid1D[2])/2;
194  if (d10*d1 <= 0) d1 = d10;
195  if (d20*d2 <= 0) d2 = d20;
196  delta = val2 - val1;
197  //Limit the derivatives for monotone data:
198  if ( d1*d2 >= 0 && d1*delta >= 0 && d2*delta >= 0){
199  if (delta == 0) return val1;
200  derivR = sqrt(pow(d1/delta, 2) + pow(d2/delta, 2));
201  if (derivR > derivLimInterp){
202  d1 *= derivLimInterp/derivR;
203  d2 *= derivLimInterp/derivR;
204  }
205  }
206  //Cubic hermite spline interpolation:
207  return pow(1-s,2)*(val1*(1 + 2*s) + d1*s) + pow(s,2)*(val2*(3 - 2*s) - d2*(1-s));
208 }
